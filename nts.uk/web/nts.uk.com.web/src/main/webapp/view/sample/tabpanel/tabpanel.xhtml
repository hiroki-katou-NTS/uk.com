API

Binding

Description
dataSource Type: array<TabModel>
Set checked value for CheckBox
active Type: string
ID string of active tab
direction
(optional)		 Type: string
Default: "horizontal"
Members:
horizontal Tabpanel display horizontal
vertical Tabpanel display vertical

TabModel

Properties

Description
id Type: string
Id for tab panel navagator
title Type: string
Title for tab panel navagator
content Type: string
Jquery selector for tab panel content
enable Type: KnockoutObservable<boolean>
Enable/disable tab panel. When disable tab panel content will be hide
visible Type: KnockoutObservable<boolean>
Show/hide tab panel navagator

TypeScript

;
	selectedTab: KnockoutObservable;

	constructor() {
		var self = this;
		self.tabs = ko.observableArray([
			{id: 'tab-1', title: 'Tab Title 1', content: '.tab-content-1', enable: ko.observable(true), visible: ko.observable(true)},

			{id: 'tab-3', title: 'Tab Title 3', content: '.tab-content-3', enable: ko.observable(true), visible: ko.observable(true)},
			{id: 'tab-4', title: 'Tab Title 4', content: '.tab-content-4', enable: ko.observable(true), visible: ko.observable(true)}
		]);
		self.selectedTab = ko.observable('tab-2');
	}
